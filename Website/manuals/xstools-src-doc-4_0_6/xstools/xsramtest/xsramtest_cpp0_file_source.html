// version info
char* version = "4.0.6";

// global variables
XSError errMsg(cerr);        // setup error channel
string lpt;                    // parallel port identifier
int portNum = -1;          // default parallel port number
string brdModel;            // XS Board model number
string ramFileFormat;        // upload RAM file format
string flashFileFormat;        // upload Flash file format
bool upload = false;      // enable reading of RAM or Flash
unsigned int loAddr, hiAddr;    // address boundaries for reading RAM
int nBitstreams = 0;       // number of FPGA/CPLD bitstream files
int posBitstreams = -1;      // position of bitstreams in argument array
int nRAMfiles = 0;           // number of RAM data files
int posRAMfiles = -1;      // position of RAM data files in argument array
int nFlashfiles = 0;       // number of Flash data files
int posFlashfiles = -1;      // position of Flash data files in argument array
bool doBatch = false;      // batch test or requires user confirmation?
bool doFullRAMTest = false;  // test every RAM location?


// begin methods and functions

// print a string that shows the options and how to use this program.
void Usage(string progName)
{
    cerr << progName.c_str() 
        << "\n\t[-[h|help]]: get help"
        << "\n\t[-[p|port] [1|2|3]]: select parallel port"
        << "\n\t[-[b|board] <type>]: select the board type to test"
//      << "\n\t[-[fpga|cpld] <files>]: download files to FPGA or CPLD"
//      << "\n\t[-[ram] <files>]: download files to RAM"
//      << "\n\t[-[flash|seeprom] <files>]: download files to flash or serial EEPROM"
//      << "\n\t[-[u|upload] <lo> <hi>]: upload RAM or Flash data between low and high addresses"
//      << "\n\t[-[f|format] [hex|mcs|exo-[16|24|32]|xess-[16|24|32]]]: file format for upload"
        << "\n";
    cerr << "Version " << version << endl;
}


// process the command line options and set the appropriate flags.
bool ProcessOpts(int* argc, char** argv)
{
    int *numFiles = NULL;   // pointer to number of FPGA/RAM/Flash files
    int *posFiles = NULL;   // pointer to position of FPGA/RAM/Flash files

    bool ok = true; // no errors yet

    for(int i=1; i< *argc; i++)
    {
        int cnt, j;
        string arg = argv[i];
        
        if(arg[0] == '-')  // found an option
        {
            ConvertToUpperCase(arg);  // convert option to upper case
            
            // ask for help but we don't give much
            if(arg=="-H" || arg=="-HELP")
            {
                Usage(arg);
                cnt = 1;
            }
            
            // specify the parallel port being used
            else if(arg=="-PORT" || arg=="-P")
            {
                sscanf(argv[i+1],"%d",&portNum);
                if(portNum<4)
                    lpt = (string)"LPT" + (string)(argv[i+1]);
                else
                    lpt = "LPT1";
                cnt = 2;
            }
            
            // enable full RAM test
            else if(arg=="-FULL")
            {
                doFullRAMTest = true;
                cnt = 1;
            }
            
            // enable batch operation
            else if(arg=="-BATCH")
            {
                doBatch = true;
                cnt = 1;
            }
            
            // specify the board model number
            else if(arg=="-BOARD" || arg=="-B")
            {
                brdModel = argv[i+1];
                ConvertToUpperCase(brdModel);
                cnt = 2;
            }

#if 0           
            // specify that the programmable logic device is to be programmed
            else if(arg=="-FPGA" || arg=="-CPLD")
            {
                if(nBitstreams > 0)
                {
                    ok = false;
                    errMsg.SetSeverity(XSErrorMinor);
                    errMsg << "cannot have more than one list of Flash/SEEPROM files" << "\n";
                    errMsg.EndMsg();
                    cnt = 1;
                }
                else
                {
                    numFiles = &nBitstreams;
                    posFiles = &posBitstreams;
                    cnt = 1;
                }
            }
            
            // specify that the RAM is to be programmed
            else if(arg=="-RAM")
            {
                if(nRAMfiles > 0)
                {
                    ok = false;
                    errMsg.SetSeverity(XSErrorMinor);
                    errMsg << "cannot have more than one list of RAM files" << "\n";
                    errMsg.EndMsg();
                    cnt = 1;
                }
                else
                {
                    numFiles = &nRAMfiles;
                    posFiles = &posRAMfiles;
                    cnt = 1;
                }
            }
            
            // specify that the Flash or serial EEPROM is to be programmed
            else if(arg=="-FLASH" || arg=="-SEEPROM")
            {
                if(nFlashfiles > 0)
                {
                    ok = false;
                    errMsg.SetSeverity(XSErrorMinor);
                    errMsg << "cannot have more than one list of Flash/SEEPROM files" << "\n";
                    errMsg.EndMsg();
                    cnt = 1;
                }
                else
                {
                    numFiles = &nFlashfiles;
                    posFiles = &posFlashfiles;
                    cnt = 1;
                }
            }

            // specify what range of XS Board RAM addresses should be uploaded into a file
            else if(arg=="-UPLOAD" || arg=="-U")
            {
                upload = true;
                sscanf(argv[i+1],"%x",&loAddr);
                // istrstream argstr1(argv[i+1]);
                // argstr1 >> hex >> loAddr;
                sscanf(argv[i+2],"%x",&hiAddr);
                // istrstream argstr2(argv[i+2]);
                // argstr2 >> hex >> hiAddr;
                cnt = 3;
            }
            
            // specify the upload file format
            else if(arg=="-FORMAT" || arg=="-F")
            {
                ramFileFormat = ConvertToUpperCase((string)(argv[i+1]));
                flashFileFormat = ConvertToUpperCase((string)(argv[i+1]));
                cnt = 2;
            }
#endif
            
            // can't figure out what the user wants
            else
            {
                ok = false;
                errMsg.SetSeverity(XSErrorMinor);
                errMsg << "unknown option: " << arg.c_str() << "\n";
                errMsg.EndMsg();
                cnt = 1;
            }
        }
        else    // found something which is not an option, so it is must be a file name
        {
            ok = false;
            errMsg.SetSeverity(XSErrorMinor);
            errMsg << "unknown option: " << arg.c_str() << "\n";
            errMsg.EndMsg();
            cnt = 1;
#if 0
            if(posFiles==NULL || numFiles==NULL)
            {
                ok = false;
                errMsg.SetSeverity(XSErrorMinor);
                errMsg << "this file is misplaced: " << arg.c_str() << "\n";
                errMsg.EndMsg();
                cnt = 1;
            }
            else
            {
                if(*posFiles<0)
                    *posFiles = i;  // record start position of list of files
                (*numFiles)++;  // increment the number of files in the list
                cnt = 0;
                i++;            // keep the file name on the argument list
            }
#endif
        }
            
        // remove the processed options from the argument list
        for(j=i+cnt; j< *argc; j++)
            argv[j-cnt] = argv[j];
        *argc -= cnt;
        i--;
    }

    return ok;
}


bool testRAM(  XSBoard* brdPtr,
                unsigned startAddr,
                unsigned numAddrBits,
                unsigned numDataBits,
                unsigned* addrBitErrs,
                unsigned* dataBitErrs,
                bool* fullPass,
                unsigned* failAddress
                )
{
    unsigned int b, data, dataRead, dataErr;
    for(b=1, dataErr=0, data=1; b<=numDataBits; b++, data<<=1)
    {
        if(brdPtr->WriteRAM(startAddr,data,BIG_ENDIAN_BYTES,LITTLE_ENDIAN_BITS))
        {
            brdPtr->WriteRAM(startAddr+1,~data,BIG_ENDIAN_BYTES,LITTLE_ENDIAN_BITS);  // purge any dynamic data on the data lines
            if(brdPtr->ReadRAM(startAddr,&dataRead,BIG_ENDIAN_BYTES,LITTLE_ENDIAN_BITS))
                dataErr |= (data ^ dataRead);
            else
                return false;
        }
        else
            return false;
    }
    *dataBitErrs = dataErr; // send back the data bit error mask
    
    // setup the data patterns for the address bit test.  Mask off any data bit errors
    // so these don't propagate through to cause erroneous address bit errors.
    unsigned int data0, data1, data2, data3, dataRead0, dataRead1, dataRead2, dataRead3;
    data0 = 0x33333333 & ~dataErr & ((1<<numDataBits)-1);
    data1 = 0x55555555 & ~dataErr & ((1<<numDataBits)-1);
    data2 = 0xAAAAAAAA & ~dataErr & ((1<<numDataBits)-1);
    data3 = 0xCCCCCCCC & ~dataErr & ((1<<numDataBits)-1);

    unsigned int addr1, addrErr;
    for(b=2, addrErr=0, addr1=2; b<=numAddrBits; b++, addr1<<=1)
    {
        unsigned int addr0;
        for(addr0=(addr1>>1); addr0>0; addr0>>=1)
        {
            bool opStatus = true;
            opStatus &= brdPtr->WriteRAM(startAddr             ,data0,      BIG_ENDIAN_BYTES,LITTLE_ENDIAN_BITS);
            opStatus &= brdPtr->WriteRAM(startAddr|      addr0 ,data1,      BIG_ENDIAN_BYTES,LITTLE_ENDIAN_BITS);
            opStatus &= brdPtr->WriteRAM(startAddr|addr1       ,data2,      BIG_ENDIAN_BYTES,LITTLE_ENDIAN_BITS);
            opStatus &= brdPtr->WriteRAM(startAddr|addr1|addr0 ,data3,      BIG_ENDIAN_BYTES,LITTLE_ENDIAN_BITS);
            opStatus &= brdPtr->ReadRAM (startAddr             ,&dataRead0, BIG_ENDIAN_BYTES,LITTLE_ENDIAN_BITS);
            opStatus &= brdPtr->ReadRAM (startAddr|      addr0 ,&dataRead1, BIG_ENDIAN_BYTES,LITTLE_ENDIAN_BITS);
            opStatus &= brdPtr->ReadRAM (startAddr|addr1       ,&dataRead2, BIG_ENDIAN_BYTES,LITTLE_ENDIAN_BITS);
            opStatus &= brdPtr->ReadRAM (startAddr|addr1|addr0 ,&dataRead3, BIG_ENDIAN_BYTES,LITTLE_ENDIAN_BITS);
            if(opStatus == false)
                return false;

            if(dataRead0 == data0)
                ;
            else if(dataRead0 == data1)
                addrErr |= addr0;
            else if(dataRead0 == data2)
                addrErr |= addr1;
            else if(dataRead0 == data3)
                addrErr |= (addr1 | addr0);
            else
                addrErr |= (addr1 | addr0);

            if(dataRead1 == data1)
                ;
            else if(dataRead1 == data0)
                addrErr |= addr0;
            else if(dataRead1 == data2)
                addrErr |= (addr1 | addr0);
            else if(dataRead1 == data3)
                addrErr |= addr1;
            else
                addrErr |= (addr1 | addr0);

            if(dataRead2 == data2)
                ;
            else if(dataRead2 == data0)
                addrErr |= addr1;
            else if(dataRead2 == data1)
                addrErr |= (addr1 | addr0);
            else if(dataRead2 == data3)
                addrErr |= addr0;
            else
                addrErr |= (addr1 | addr0);

            if(dataRead3 == data3)
                ;
            else if(dataRead3 == data0)
                addrErr |= (addr1 | addr0);
            else if(dataRead3 == data1)
                addrErr |= addr1;
            else if(dataRead3 == data2)
                addrErr |= addr0;
            else
                addrErr |= (addr1 | addr0);
        }
    }

    *addrBitErrs = addrErr;

    *fullPass = true;
    if(doFullRAMTest)
    {
        unsigned int a, d, mask;
        mask = (1<<numDataBits)-1;
        for(a=startAddr; a<(startAddr+(1<<numAddrBits)); a++)
        {
            if(!brdPtr->WriteRAM(a,a,BIG_ENDIAN_BYTES,LITTLE_ENDIAN_BITS))
                return false;
        }
        for(a=startAddr; a<(startAddr+(1<<numAddrBits)); a++)
        {
            if(!brdPtr->ReadRAM(a,&d,BIG_ENDIAN_BYTES,LITTLE_ENDIAN_BITS))
                return false;
            if((d&mask)!=(a&mask))
            {
                *failAddress = a;
                *fullPass = false;
                return true;
            }
        }
    }

    return true;
}

int main( int argc, char **argv )
{
    // get the board type used in the most recent invocation of an XSTOOLS utility
    brdModel = GetXSTOOLSParameter("BoardType");

#if 0
    // get the RAM hex file type used in the most recent invocation of an XSTOOLS utility
    ramFileFormat = GetXSTOOLSParameter("RAMFormat");

    // get the Flash hex file type used in the most recent invocation of an XSTOOLS utility
    flashFileFormat = GetXSTOOLSParameter("FlashFormat");
#endif

    // get the parallel port used in the most recent invocation of an XSTOOLS utility
    lpt = GetXSTOOLSParameter("LPT");
    if(lpt != "")
    {
        int n = sscanf(lpt.c_str()+3,"%d",&portNum);
        assert(n != 0);
    }

    // process command-line options
    if(ProcessOpts(&argc,argv) == false)
    {
        Usage(string(argv[0]));
        return 1;   // exit if error in program options
    }

    if(doBatch)
    {
        errMsg.EnableBatch(true);    // turn off error messages
        EnableBatch(true);     // turn off user prompts
    }

    // it's an error if the board model being tested is not specified
    if(brdModel == "")
    {
        errMsg.SimpleMsg(XSErrorMajor,"No XS Board model was specified!\n");
        Usage(string(argv[0]));
        return 1;
    }

    // store the model type of the board that was specified
    SetXSTOOLSParameter("BoardType",brdModel.c_str());

    // use LPT1 if no other parallel port was specified
    if(portNum == -1)
    {
        portNum = 1;
        lpt = "LPT1";
    }

    // store the parallel port that was specified
    if(portNum < 4)
        SetXSTOOLSParameter("LPT",lpt.c_str());

    // determine the type of XS Board and set the pointer to the board object
    XSBoard* brdPtr;
    brdPtr = NewXSBoard(brdModel.c_str());
    if(brdPtr==NULL)
    {
        errMsg.SimpleMsg(XSErrorMajor,"Unknown type of XS Board was specified!\n");
        return 1;
    }

    // initialize the board object
    if(brdPtr->Setup(&errMsg,brdModel.c_str(),portNum) == false)
    {
        errMsg.SimpleMsg(XSErrorMajor,"Invalid parallel port was selected!\n");
        delete brdPtr;
        return 1;
    }

    unsigned addrBitErrs, dataBitErrs, failAddress;
    bool fullPass;
    if(strncmp(brdModel.c_str(),"XSA",strlen("XSA"))==0)
    {
        if(testRAM(brdPtr,0,23,16,&addrBitErrs,&dataBitErrs,&fullPass,&failAddress) == false)
        {
            errMsg.SimpleMsg(XSErrorMajor,"Error reading/writing RAM during test!\n");
            delete brdPtr;
            return 1;
        }

        unsigned mask;
        int i;
        printf( "Data bit errors =" );
        for(mask=1<<31, i=0; mask>0; mask>>=1, i++)
        {
            if(i%4==0) printf( " " );
            printf( "%c", (dataBitErrs & mask) ? 'X':'-' );
        }
        printf( "\nAddress bit errors =" );
        for(mask=1<<31, i=0; mask>0; mask>>=1, i++)
        {
            if(i%4==0) printf( " " );
            printf( "%c", (addrBitErrs & mask) ? 'X':'-' );
        }
        printf("\n");
        if(!fullPass)
        {
            printf( "Failure at RAM address: %x\n", failAddress);
        }
        
        delete brdPtr;
        
        return (addrBitErrs==0 && dataBitErrs==0) ? 0 : 1;  // 0 indicates a successful download/upload
    }
    else if(strncmp(brdModel.c_str(),"XSB",strlen("XSB"))==0)
    {
        if(testRAM(brdPtr,0,18,16,&addrBitErrs,&dataBitErrs,&fullPass,&failAddress) == false)
        {
            errMsg.SimpleMsg(XSErrorMajor,"Error reading/writing RAM during test!\n");
            delete brdPtr;
            return 1;
        }

        unsigned mask;
        int i;
        printf( "Data bit errors =" );
        for(mask=1<<31, i=0; mask>0; mask>>=1, i++)
        {
            if(i%4==0) printf( " " );
            printf( "%c", (dataBitErrs & mask) ? 'X':'-' );
        }
        printf( "\nAddress bit errors =" );
        for(mask=1<<31, i=0; mask>0; mask>>=1, i++)
        {
            if(i%4==0) printf( " " );
            printf( "%c", (addrBitErrs & mask) ? 'X':'-' );
        }
        printf("\n");
        if(!fullPass)
        {
            printf( "Failure at RAM address: %x\n", failAddress);
        }
        
        delete brdPtr;
        
        return (addrBitErrs==0 && dataBitErrs==0) ? 0 : 1;  // 0 indicates a successful download/upload
    }
    else if(strncmp(brdModel.c_str(),"XSV",strlen("XSV"))==0)
    {
        bool opStatus, leftFullPass, rightFullPass;
        unsigned leftAddrBitErrs, leftDataBitErrs, rightAddrBitErrs, rightDataBitErrs, leftFailAddress, rightFailAddress;
        opStatus = testRAM(brdPtr,0,19,16,&rightAddrBitErrs,&rightDataBitErrs,&rightFullPass,&rightFailAddress);
        opStatus &= testRAM(brdPtr,1<<19,19,16,&leftAddrBitErrs,&leftDataBitErrs,&leftFullPass,&leftFailAddress);
        if(opStatus == false)
        {
            errMsg.SimpleMsg(XSErrorMajor,"Error reading/writing RAM during test!\n");
            delete brdPtr;
            return 1;
        }

        unsigned mask;
        int i;
        printf( "Left-bank data bit errors =" );
        for(mask=1<<31, i=0; mask>0; mask>>=1, i++)
        {
            if(i%4==0) printf( " " );
            printf( "%c", (leftDataBitErrs & mask) ? 'X':'-' );
        }
        printf( "\nLeft-bank address bit errors =" );
        for(mask=1<<31, i=0; mask>0; mask>>=1, i++)
        {
            if(i%4==0) printf( " " );
            printf( "%c", (leftAddrBitErrs & mask) ? 'X':'-' );
        }
        printf("\n");
        if(!leftFullPass)
        {
            printf( "Failure at left-bank RAM address: %x\n", leftFailAddress);
        }
        printf( "Right-bank data bit errors =" );
        for(mask=1<<31, i=0; mask>0; mask>>=1, i++)
        {
            if(i%4==0) printf( " " );
            printf( "%c", (rightDataBitErrs & mask) ? 'X':'-' );
        }
        printf( "\nRight-bank address bit errors =" );
        for(mask=1<<31, i=0; mask>0; mask>>=1, i++)
        {
            if(i%4==0) printf( " " );
            printf( "%c", (rightAddrBitErrs & mask) ? 'X':'-' );
        }
        printf("\n");
        if(!rightFullPass)
        {
            printf( "Failure at right-bank RAM address: %x\n", rightFailAddress);
        }
        
        delete brdPtr;
        
        return (rightAddrBitErrs==0 && rightDataBitErrs==0 && leftAddrBitErrs==0 && leftDataBitErrs==0) ? 0 : 1;    // 0 indicates a successful download/upload
    }
    else if(
        strncmp(brdModel.c_str(),"XS95",strlen("XS95"))==0 ||
        strncmp(brdModel.c_str(),"XS40",strlen("XS40"))==0 ||
        strncmp(brdModel.c_str(),"XSP",strlen("XSP"))==0
        )
    {
        if(testRAM(brdPtr,0,17,8,&addrBitErrs,&dataBitErrs,&fullPass,&failAddress) == false)
        {
            errMsg.SimpleMsg(XSErrorMajor,"Error reading/writing RAM during test!\n");
            delete brdPtr;
            return 1;
        }

        unsigned mask;
        int i;
        printf( "Data bit errors =" );
        for(mask=1<<31, i=0; mask>0; mask>>=1, i++)
        {
            if(i%4==0) printf( " " );
            printf( "%c", (dataBitErrs & mask) ? 'X':'-' );
        }
        printf( "\nAddress bit errors =" );
        for(mask=1<<31, i=0; mask>0; mask>>=1, i++)
        {
            if(i%4==0) printf( " " );
            printf( "%c", (addrBitErrs & mask) ? 'X':'-' );
        }
        printf("\n");
        if(!fullPass)
        {
            printf( "Failure at RAM address: %x\n", failAddress);
        }
        
        delete brdPtr;
        
        return (addrBitErrs==0 && dataBitErrs==0) ? 0 : 1;  // 0 indicates a successful download/upload
    }

    delete brdPtr;

    return 0;   // 0 indicates a successful download/upload
}
