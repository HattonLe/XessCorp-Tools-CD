int map_pin_to_bit(unsigned int pin)
{
    int map[] = {
            -1, // illegal
            16, // C0
            0,  // D0
            1,  // D1
            2,  // D2
            3,  // D3
            4,  // D4
            5,  // D5
            6,  // D6
            7,  // D7
            14, // S6
            15, // S7
            13, // S5
            12, // S4
            17, // C1
            11, // S3
            18, // C2
            19, // C3
    };

    if(pin<1 || pin>sizeof(map)/sizeof(int))
    {
        err.SetSeverity(XSErrorFatal);
        err << "Illegal parallel port pin was specified!!\n";
        err.EndMsg();
    }

    return map[pin];
}
