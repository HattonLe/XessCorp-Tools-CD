int main(int argc, char **argv)
{
    // Setup error channel.
    XSError err(cerr);

    lpt = GetXSTOOLSParameter("LPT");
    if(lpt != "")
    {
        int n = sscanf(lpt.c_str()+3,"%d",&portNum);
        assert(n != 0);
    }
    
    if(ProcessOpts( &argc, argv ) == false)
    {
        Usage(string(argv[0]));
        return 1;   // exit if error in program options
    }

    // Use LPT1 if no other parallel port was specified.
    if(portNum == -1)
    {
        portNum = 1;
        lpt = "LPT1";
    }

    // Store the parallel port that will be used.
    if(portNum < 4)
        SetXSTOOLSParameter("LPT",lpt.c_str());

    if(argc<2)
        return 1;   // exit if no binary string of bits

    // Create the parallel port object.
    PPort port;
    if(port.Setup(&errMsg,portNum,0x3) == false)
    {
        err.SimpleMsg(XSErrorMajor,"Invalid parallel port was selected!\n");
        return 1;
    }
    
    // Iterate through the string while assigning binary values to the parallel port data bits. 
    int portValue = port.In(0,7); // get current value on parallel port
    string bits = argv[1];      // get the string of binary bits
    int i = bits.length()-1;
    for( unsigned int bitMask=1; (i>=0) && (bitMask!=256); bitMask<<=1,i-- )
    {
        switch( bits[i] )
        {
            // binary one is represented by '1', 'h', or 'H'
        case '1':
        case 'H':
        case 'h':
            portValue |= bitMask;
            break;
            // binary zero is represented by '0', 'l', or 'L'
        case '0':
        case 'L':
        case 'l':
            portValue &= ~bitMask;
            break;
            // erroneous binary value if we get here!!
        default:
            errMsg.SetSeverity(XSErrorMinor);
            errMsg << "illegal binary value in " << bits.c_str() << "\n";
            errMsg.EndMsg();
            break;
        }
    }
    
    // Exit without changing the parallel port if there was an error.
    if(errMsg.IsError())
        return 1;   // return 1 if something went wrong

    // output the new value on the parallel port
    port.Out(portValue,0,7);

    if(getStatus)
        cout << hex << port.In(11,15) << "\n";

    return 0;   // return 0 if the operation was successful
}
