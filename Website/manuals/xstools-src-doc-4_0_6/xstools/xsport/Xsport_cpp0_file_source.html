/**
\unit
\file
File containing the main routine and supporting subroutines for the command-line program that forces bit values onto pins of the XESS boards.
*/


#include <cstdlib>
#include <cassert>
#include <sstream>
#include <fstream>
#include <string>
using namespace std;

#include "xserror.h"
#include "pport.h"
#include "utils.h"
char* version = "4.0.6"; ///< Version info
string lpt;            ///< Parallel port identifier
int portNum = -1;  ///< Parallel port number
bool getStatus = false;  ///< Get value of parallel port status bits
XSError errMsg(cerr); ///< Setup error-reporting channel
bool writeControlBits = false; ///< True if bitstring is destined for control bits


/// Print a string that shows the options and how to use this program.
void Usage(string progName)
{
    cerr << progName.c_str() << " [-[h|help]] [-[p|port] [1|2|3]] [-[s|status]] <binary number>" << endl;
    cerr << "Version " << version << endl;
}


/// Process the command line options and set the appropriate flags.
bool ProcessOpts(int* argc, char** argv)
{
    XSError err(cerr); // setup error channel

    bool ok = true; // no error yet

    for(int i=1; i< *argc; i++)
    {
        int cnt, j;
        string arg = argv[i];
        
        if(arg[0] == '-')  // found an option
        {
            ConvertToUpperCase(arg);  // convert option to upper case
            
            // ask for help but we don't give much
            if(arg=="-H" || arg=="-HELP")
            {
                Usage(arg);
                cnt = 1;
            }
            
            // specify the printer port being used
            else if(arg=="-PORT" || arg=="-P")
            {
                sscanf(argv[i+1],"%d",&portNum);
                // istrstream argstr(argv[i+1]);
                // argstr >> portNum;
                if(portNum<4)
                    lpt = (string)"LPT" + (string)(argv[i+1]);
                else
                    lpt = "LPT1";
                cnt = 2;
            }

            // ask for the value of the parallel port status bits
            else if(arg=="-STATUS" || arg=="-S")
            {
                getStatus = true;
                cnt = 1;
            }
            
            // can't figure out what the user wants
            else
            {
                ok = false;
                err.SetSeverity(XSErrorMinor);
                err << "unknown option: " << arg.c_str() << "\n";
                err.EndMsg();
                cnt = 1;
            }
        }
        else    // found something which is not an option, so it must be a bit string
        { // skip over this argument and leave it in the argument array
            i++;
            cnt = 0;
        }
            
        // remove the processed options from the argument list
        for(j=i+cnt; j< *argc; j++)
            argv[j-cnt] = argv[j];
        *argc -= cnt;
        i--;
    }

    return ok;
}


/// Main routine.
/// \return 0 if the bits are output successfully; 1 if some problem occurs.
int main( int argc, char **argv )
{
    // Setup error channel.
    XSError err(cerr);

    lpt = GetXSTOOLSParameter("LPT");
    if(lpt != "")
    {
        int n = sscanf(lpt.c_str()+3,"%d",&portNum);
        assert(n != 0);
    }
    
    if(ProcessOpts( &argc, argv ) == false)
    {
        Usage(string(argv[0]));
        return 1;   // exit if error in program options
    }

    // Use LPT1 if no other parallel port was specified.
    if(portNum == -1)
    {
        portNum = 1;
        lpt = "LPT1";
    }

    // Store the parallel port that will be used.
    if(portNum < 4)
        SetXSTOOLSParameter("LPT",lpt.c_str());

    if(argc<2)
        return 1;   // exit if no binary string of bits

    // Create the parallel port object.
    PPort port;
    if(port.Setup(&errMsg,portNum,0x3) == false)
    {
        err.SimpleMsg(XSErrorMajor,"Invalid parallel port was selected!\n");
        return 1;
    }
    
    // Iterate through the string while assigning binary values to the parallel port data bits. 
    int portValue = port.In(0,7); // get current value on parallel port
    string bits = argv[1];      // get the string of binary bits
    int i = bits.length()-1;
    for( unsigned int bitMask=1; (i>=0) && (bitMask!=256); bitMask<<=1,i-- )
    {
        switch( bits[i] )
        {
            // binary one is represented by '1', 'h', or 'H'
        case '1':
        case 'H':
        case 'h':
            portValue |= bitMask;
            break;
            // binary zero is represented by '0', 'l', or 'L'
        case '0':
        case 'L':
        case 'l':
            portValue &= ~bitMask;
            break;
            // erroneous binary value if we get here!!
        default:
            errMsg.SetSeverity(XSErrorMinor);
            errMsg << "illegal binary value in " << bits.c_str() << "\n";
            errMsg.EndMsg();
            break;
        }
    }
    
    // Exit without changing the parallel port if there was an error.
    if(errMsg.IsError())
        return 1;   // return 1 if something went wrong

    // output the new value on the parallel port
    port.Out(portValue,0,7);

    if(getStatus)
        cout << hex << port.In(11,15) << "\n";

    return 0;   // return 0 if the operation was successful
}
