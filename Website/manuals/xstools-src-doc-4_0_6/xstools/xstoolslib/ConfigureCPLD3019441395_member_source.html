void XC95KPort::ConfigureCPLD(unsigned char b)
{
    // JTAG configuration bytes contain values for both TDI and TMS,
    // so only four clock pulses are needed per byte
    for(int i=0; i<4; i++)
    {
        SetTDI(b&1);  // byte is sent out LSB first
        b >>= 1;
        SetTMS(b&1);
        b >>= 1;
        PulseTCK();
    }
}
