int PromptUser(string &msg, int action)
{
    if(batch)
        return RESPONSE_CONTINUE;

#ifdef _WINDOWS
    int response;
    switch(action)
    {
    case PROMPT_OKCANCEL:
        response = AfxMessageBox(msg.c_str(),MB_ICONINFORMATION | MB_OKCANCEL);
        return response==IDCANCEL ? RESPONSE_CANCEL : RESPONSE_CONTINUE;
    case PROMPT_OK:
    default:
        AfxMessageBox(msg.c_str(),MB_ICONINFORMATION);
        return RESPONSE_CONTINUE;
    }
#else
    cerr << msg.c_str();
    char c;
    switch(action)
    {
    case PROMPT_OKCANCEL:
        cerr << "\nContinue (Y/N)?...";
        c = toupper(getchar());
        fflush(stdin);
        cerr << "\n";
        if(c == 'Y')
            return RESPONSE_CONTINUE;
        else
            return RESPONSE_CANCEL;
    case PROMPT_OK:
    default:
        cerr << "\nPress any key to continue...";
        getchar();
        fflush(stdin);
        cerr << "\n";
        return RESPONSE_CONTINUE;
    }
#endif
}
