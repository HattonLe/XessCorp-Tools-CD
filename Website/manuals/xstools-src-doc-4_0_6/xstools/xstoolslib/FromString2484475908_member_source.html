void Bitstream::FromString(const string s)
{
    Resize(s.length());
    // the last char in the string is the LSB of the bitstream
    for(int i=numBits-1, j=0; i>=0; i--)
    {
        if(isspace(s[i]))
            continue;   // skip whitespace
        this->SetBit(j++, s[i]=='0' ? 0:1);
    }
}
