bool LPTJTAG::Setup(XSError *e, unsigned int portNum, unsigned int invMask, unsigned int pos_tck, unsigned int pos_tms, unsigned int pos_tdi, unsigned int pos_tdo)
{
    posTCK = pos_tck;
    posTMS = pos_tms;
    posTDI = pos_tdi;
    posTDO = pos_tdo;
    SetTraceOnOff(false,cerr);   // don't trace TAP signals
    JTAGPort::Setup(e);

}
