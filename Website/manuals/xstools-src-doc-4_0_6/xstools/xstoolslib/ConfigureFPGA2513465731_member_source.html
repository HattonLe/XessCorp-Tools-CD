void XCVPort::ConfigureFPGA(unsigned char b)
{
    if(fastDownload)
    { // fast parallel configuration
        // reverse the bits of the configuration byte
        unsigned char rev_b;
        for(int i=0; i<8; i++)
            rev_b = (rev_b<<1) | ((b>>i) & 0x1);
        Out(rev_b,posDLO,posDHI);
        SetCCLK(1);
        SetCCLK(0);
    }
    else
    { // slow serial configuration
        // configuration bytes for the XC4000 programming pins only contain
        // values for the DIN pin, so eight clock pulses are needed per byte.
        for(int i=0; i<8; i++)
        {
            SetDIN(b&0x80 ? 1:0);  // byte is sent MSB first
            b <<= 1;
            PulseCCLK();
        }
    }
}
