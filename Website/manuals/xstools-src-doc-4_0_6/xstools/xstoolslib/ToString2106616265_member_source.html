string Bitstream::ToString(void) const
{
    // start printing with the MSB which is stored at the
    // end of the bitstream.  Then work our way to the beginning.
    string s = "";
    for(int i=numBits; i>0; )
        s += ((*this)[--i]==1 ? "1":"0");
    return s;
}
