void NextCmd(istream &is)
{
    char c, waste[257];
    while(!is.fail())
    {
        is.read(&c,1);
        if(is.fail())
            break;
        switch(c)
        {
        case '/':
            if(is.peek()=='/')
                is.getline(waste,256);
            break;

        case '!':
            is.getline(waste,256);
            break;

        case ';':
            return;

        default:
            break;
        }
    }
}
