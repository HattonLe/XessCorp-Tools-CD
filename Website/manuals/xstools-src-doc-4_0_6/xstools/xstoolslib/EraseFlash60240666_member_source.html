bool F28Port::EraseFlash(void)
{
    string desc("Flash Erase"), subdesc("Erasing flash device");
    Progress* progressGauge = new Progress(NULL,desc,subdesc,0,sizeof(flashBlocks)/sizeof(struct flashBlock)-1);
    for(int i=0; i<sizeof(flashBlocks)/sizeof(struct flashBlock); i++)
    {
        if(EraseFlashBlock(i) == false)
        {
            delete progressGauge;
            return false;
        }
        progressGauge->Report(i);
    }
    delete progressGauge;
    return true;
}
