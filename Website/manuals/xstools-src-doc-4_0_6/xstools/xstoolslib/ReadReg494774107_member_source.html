void I2CPort::ReadReg(unsigned char devAddr, unsigned char regAddr, unsigned char *regData)
{
    unsigned char buf[3];
    buf[0] = devAddr & ~0x01;   // create write address
    buf[1] = regAddr;
    buf[2] = devAddr | 0x01;    // create read address
    ReadBuffer(3,buf,1,regData);
}
