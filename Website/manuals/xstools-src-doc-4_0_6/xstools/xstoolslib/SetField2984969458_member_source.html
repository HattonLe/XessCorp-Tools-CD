unsigned int SetField(unsigned int data, unsigned int loPos, unsigned int hiPos, unsigned int newData)
{
    assert(loPos<=maxBitPos);
    assert(hiPos<=maxBitPos);
    assert(hiPos>=loPos);
    return (data & ~FIELDMASK(loPos,hiPos)) |
        ((newData & FIELDMASK(0,hiPos-loPos))<<loPos);
}
