void MchpPort::EnterICSP(void)
{
    // Set all the ICSP inputs to 0
    SetPGC(0);
    SetPGD(0);
    SetMCLR_N(0);
    SetPGM(0);

    // Raise PGM and then MCLR_N to start ICSP
    SetPGM(1);
    InsertDelay(TIME_P15);
    SetMCLR_N(1);
    InsertDelay(TIME_P12);
//  InsertDelay(200,MILLISECONDS);
}
