unsigned char PPort::Inp(unsigned short regOffset) const
{
    assert(regOffset < 3);
    switch(IODriverIndex)
    {
    case DRIVERLINX:
        return DlPortReadPortUchar(address + regOffset);
        break;
    case TVICHW32:
        return GetPortByte(PPort::HW32, address + regOffset);
        break;
    case UNIIO:
    default:
        switch(regOffset)
        {
        case DATAREG:
            return dataPort->readChar();
            break;
        case STATREG:
            return statusPort->readChar();
            break;
        case CTRLREG:
            return controlPort->readChar();
            break;
        default:
            assert(1==0);
            break;
        }
        break;
    }
    return 0;
}
