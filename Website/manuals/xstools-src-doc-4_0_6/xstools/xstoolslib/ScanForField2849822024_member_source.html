bool ScanForField(istream &is, unsigned char searchType)
{
    while(true)
    {
        unsigned char type = GetType(is);
        if(type==searchType)
            return true;
        unsigned int length = GetInteger(is);
        is.ignore(length);
        if(is.eof()!=0)
            break;
    }
    return false;
}
