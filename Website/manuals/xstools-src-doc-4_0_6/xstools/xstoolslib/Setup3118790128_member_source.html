bool JTAGRAMPort::Setup(int xc95Type, XSError *e, unsigned int portNum, unsigned int invMask, unsigned int pos_tck, unsigned int pos_tms, unsigned int pos_tdi, unsigned int pos_tdo)
{
    if(bsirPtr != NULL)
        delete bsirPtr;
    if(bsdrPtr != NULL)
        delete bsdrPtr;
    bsirPtr = new Bitstream(bsirLength[xc95Type]);
    assert(bsirPtr != NULL);
    bsdrPtr = new XCBSDR(bsdrLength[xc95Type],
                        ramAddressBitPos[xc95Type],
                        ramDataBitPos[xc95Type],
                        ramControlBitPos[xc95Type]);
    assert(bsdrPtr != NULL);
    return LPTJTAG::Setup(e,portNum,invMask,pos_tck,pos_tms,pos_tdi,pos_tdo);
}
