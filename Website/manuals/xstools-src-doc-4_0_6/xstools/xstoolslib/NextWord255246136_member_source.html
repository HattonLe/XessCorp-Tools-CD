string NextWord(istream &is)
{
    char c, waste[257];
    string word = "";
    while(!is.fail())
    {
        is.read(&c,1);
        if(is.fail())
            break;
        switch(c)
        {
        case '/':
            if(is.peek()=='/')
            {
                is.getline(waste,256);
                if(word.length() != 0)
                    return word;
            }
            else
                word = word + c;
            break;

        case '!':
            is.getline(waste,256);
            if(word.length() != 0)
                return word;
            break;

        case '\n':
            break;

        case ' ':
        case '\t':
            if(word.length() != 0)
                return word;
            break;

        case ';':
            is.putback(c);
            return word;

        default:
            word = word + c;
            break;
        }
    }
    return word;
}
