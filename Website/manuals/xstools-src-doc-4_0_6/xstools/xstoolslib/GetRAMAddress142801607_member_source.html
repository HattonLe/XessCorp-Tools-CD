unsigned long XCBSDR::GetRAMAddress(void)
{
    unsigned int address = 0;
    for(int i=numRAMAddressBits-1; i>=0; i--)
        address = (address<<1) | ((*this)[ramAddressBitPos[i]+bsdrPinInput] ? 1:0);
    return address;
