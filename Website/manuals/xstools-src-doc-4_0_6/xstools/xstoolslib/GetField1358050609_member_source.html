unsigned int GetField(unsigned int data, unsigned int loPos, unsigned int hiPos)
{
    assert(loPos<=maxBitPos);
    assert(hiPos<=maxBitPos);
    assert(hiPos>=loPos);
    return (FIELDMASK(loPos,hiPos) & data) >> loPos;
}
