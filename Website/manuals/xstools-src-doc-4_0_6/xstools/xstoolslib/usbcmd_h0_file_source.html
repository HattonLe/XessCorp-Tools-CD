#ifndef USBCMD_H
#define USBCMD_H

/**
Definitions of commands sent in USB packets to the XSUSB interface.
*/
enum USBCmd
{
    READ_VERSION            = 0x00,
    /*
    READ_FLASH              = 0x01,
    WRITE_FLASH             = 0x02,
    ERASE_FLASH             = 0x03,
    READ_EEDATA             = 0x04,
    WRITE_EEDATA            = 0x05,
    READ_CONFIG             = 0x06,
    WRITE_CONFIG            = 0x07,
    */
    ID_BOARD                = 0x31,
    UPDATE_LED            = 0x32,
    INFO_CMD                = 0x40, // Get information about the USB interface.
    SET_PORT_BIT_INVERSIONS    = 0x42, // Set inversion mask for bits on the JTAG port 
    TMS_TDI_CMD                = 0x43, // Send a single TMS and TDI bit.
    TMS_TDI_TDO_CMD            = 0x44, // Send a single TMS and TDI bit and receive TDO bit.
    TDI_TDO_CMD                = 0x46, // Send multiple TDI bits and receive multiple TDO bits.
    TDI_CMD                    = 0x48, // Send multiple TDI bits.
    RUNTEST_CMD                = 0x49, // Pulse TCK a given number of times.
    NULL_TDI_CMD            = 0x50, // Send string of TDI bits.
    RESET                = 0xff
};

#endif
