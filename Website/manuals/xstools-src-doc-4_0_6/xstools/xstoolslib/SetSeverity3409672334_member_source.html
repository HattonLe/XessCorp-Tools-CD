void XSError::SetSeverity(XSErrorSeverity s)
{
    severity = s;
    switch(severity)
    {
    case XSErrorFatal:
        *os << GetHeader().c_str() << " FATAL ERROR: ";
        SetState(XSErrorInMessage);
        break;
    case XSErrorMajor:
        *os << GetHeader().c_str() << " MAJOR ERROR: ";
        os->flush();
        SetState(XSErrorInMessage);
        break;
    case XSErrorMinor:
        *os << GetHeader().c_str() << " MINOR ERROR: ";
        os->flush();
        SetState(XSErrorInMessage);
        break;
    case XSErrorNone:
        *os << GetHeader().c_str() << ": ";
        os->flush();
        SetState(XSErrorInMessage);
        break;
    default:
        SetSeverity(XSErrorMinor);
        *os << "\nerror severity was incorrectly set!\n";
        os->flush();
        EndMsg();
        break;
    }
}
