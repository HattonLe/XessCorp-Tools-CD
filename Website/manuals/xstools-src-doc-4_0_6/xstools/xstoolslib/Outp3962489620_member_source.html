void PPort::Outp(unsigned short regOffset, unsigned char byte)
{
    assert(regOffset < 3);
    switch(IODriverIndex)
    {
    case DRIVERLINX:
        DlPortWritePortUchar(address + regOffset, byte);
        break;
    case TVICHW32:
        SetPortByte(PPort::HW32, address + regOffset, byte);
        break;
    case UNIIO:
    default:
        switch(regOffset)
        {
        case DATAREG:
            dataPort->write(byte);
            break;
        case STATREG:
            statusPort->write(byte);
            break;
        case CTRLREG:
            controlPort->write(byte);
            break;
        default:
            assert(1==0);
            break;
        }
        break;
    }
}
