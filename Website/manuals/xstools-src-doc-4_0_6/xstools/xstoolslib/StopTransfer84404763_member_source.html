void I2CPort::StopTransfer()
{
    assert(GetSCL()==0);
    SetSDA(0);        // make sure data line is low
    InsertDelay(bitDur,MICROSECONDS);
    SetSCL(1);        // raise clock
    InsertDelay(bitDur,MICROSECONDS);
    SetSDA(1);        // raise data while clock is high to stop I2C transfer
    InsertDelay(bitDur,MICROSECONDS);
}
